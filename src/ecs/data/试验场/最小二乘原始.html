def shape(A):
    assert(type(A)==list)
    B = A
    s = []
    while(type(B)==list and len(B)!=0):
        s.append(len(B))
        B = B[0]

def zeros(s,dtype=float):
    if type(s)==int:
        return [0 for _ in range(s)]
    elif type(s) == tuple and len(s)==1:
        return [0 for _ in range(s[0])]
    else:
        r = zeros(s[1:],dtype=dtype)
        result = []
        for i in range(s[0]):
            result.append(r)
    return result

def identity_matrix(s,dtype=float):
    assert(type(s)==int)
    result = []
    for i in range(s):
        r = []
        for j in range(s):
            if i==j:
                r.append(dtype(1))
            else:
                r.append(dtype(0))
        result.append(r)
    return result

def matrix_inverse(A):
    assert(dim(A)==2)
    N = shape(A)[0]
    R = identity_matrix(N)
    #     
    for j in range(N):
        for i in range(N):
            if i>j:
                fa = -A[i][j]/A[j][j]
                row_assign(A,i,j,fa)
                row_assign(R,i,j,fa)

    #     
    for j in range(N)[::-1]:
        for i in range(N)[::-1]:
            if i<j:
                fa = -A[i][j]/A[j][j]
                row_assign(A,i,j,fa)
                row_assign(R,i,j,fa)
                
    for i in range(len(R)):
        R[i] = [x / A[i][i] for x in R[i]]
    return R

def matrix_transpose(A):
    assert(dim(A)==2)
    m = shape(A)[0]
    n = shape(A)[1]
    result = []
    for j in range(n):
        r = []
        for i in range(m):
            r.append(A[i][j])
        result.append(r)
    return result

def matrix_mutmul(A,B):
    assert(dim(A)==2 and dim(B)==2 and shape(A)[1]==shape(B)[0])
    def __sub_product(A,i,B,j):
        N = len(A[i])
        partial_sum = 0
        for k in range(N):
            partial_sum += A[i][k]*B[k][j]
        return partial_sum
    
    m = shape(A)[0]
    n = shape(B)[1]
    
    R = []
    for i in range(m):
        r = []
        for j in range(n):
            r.append(__sub_product(A,i,B,j))
        R.append(r)
    return R
